001package software.leonov.io;
002
003
004import java.util.function.Consumer;
005
006/**
007 * A utility class for working with {@code AutoCloseable} instances.
008 *
009 * @author Zhenya Leonov
010 */
011public final class AutoCloseables {
012
013    private AutoCloseables() {
014    };
015
016    /**
017     * Closes an {@link AutoCloseable} resource, with explicit control over how to handle any error that may occur.
018     * <p>
019     * This is primarily useful when manually cleaning up resources, where a thrown exception needs to be logged,
020     * suppressed, or otherwise handled without being propagated.
021     * <p>
022     * <b>Discussion:</b> Common {@code closeQuietly} implementations generally fall into two categories:
023     * </p>
024     * <ul>
025     * <li>They silently ignore any {@link Exception} that occurs during the closing process. While seemingly convenient,
026     * this can mask critical errors, especially when writing to an I/O resource where data loss or corruption might occur
027     * without any indication.</li>
028     * <li>They log any closing exception using {@code java.util.logging}. This approach is preferable because it
029     * acknowledges the error, but it lacks flexibility, as the user has no direct control over the logger configuration or
030     * the handling of the logged exception.</li>
031     * </ul>
032     * <p>
033     * In contrast, this method allows the user to provide a {@link Consumer} that will be invoked with any {@link Throwable
034     * error} thrown by the {@code close()} method. This enables you to implement custom error logging, reporting, or
035     * recovery mechanisms tailored to your specific application needs.
036     * 
037     * @param resource the {@code AutoCloseable} resource to close or {@code null} in which case this method is a no-op
038     * @param consumer the exception handler to use
039     */
040    public static void closeQuietly(final AutoCloseable resource, final Consumer<Throwable> consumer) {
041        if (resource != null)
042            try {
043                resource.close();
044            } catch (final Throwable t) {
045                consumer.accept(t);
046            }
047    }
048
049}
