001package software.leonov.util;
002
003import static java.util.Objects.requireNonNull;
004
005import java.util.ArrayDeque;
006import java.util.Deque;
007
008/**
009 * An {@link AutoCloseable} that collects {@code AutoCloseable} resources and closes them all when it is {@link #close
010 * closed}. This class is the {@code AutoCloseable} analog to Guava's
011 * <a href="https://guava.dev/releases/26.0-jre/api/docs/index.html?com/google/common/io/Closer.html" target=
012 * "_blank">Closer</a> class. See Guava's <a href="https://github.com/google/guava/issues/3450" target="_blank">Issue
013 * 3450</a>, <a href="https://github.com/google/guava/issues/3068" target="_blank">Issue 3068</a>, and
014 * <a href="https://github.com/google/guava/issues/1020" target="_blank">Issue 1020</a> for further discussion.
015 * <p>
016 * {@code AutoCloseable} class has two main use cases:
017 * <ul>
018 * <li>When the number of {@code AutoCloseable} resources is not known until runtime (e.g. the resources are user
019 * supplied).</li>
020 * <li>When properly closing all {@code AutoCloseable} resources requires nested
021 * <a href="http://docs.oracle.com/javase/tutorial/essential/exceptions/tryResourceClose.html" target=
022 * "_blank">try-with-resources</a> blocks which make the code too cumbersome.</li>
023 * </ul>
024 * <p>
025 * This class is intended to be used in the following pattern:
026 * <p>
027 * Java 6 style:
028 * 
029 * <pre><code class="line-numbers match-braces language-java">
030 * final AutoCloser closer = new AutoCloser();
031 * try {
032 *     final InputStream in = closer.register(...);
033 *     final OutputStream out = closer.register(...);
034 *     ...
035 * } catch (final Throwable t) {
036 *     throw closer.rethrow(t);
037 * } finally {
038 *     closer.close();
039 * }
040 * </code></pre>
041 * 
042 * Java 7+ try-with-resources style:
043 * 
044 * <pre><code class="line-numbers match-braces language-java">
045 * try (final AutoCloser closer = new AutoCloser()) {
046 *     final InputStream in = closer.register(...);
047 *     final OutputStream out = closer.register(...);
048 *     ...
049 * }
050 * </code></pre>
051 *
052 * @author Zhenya Leonov
053 */
054public final class AutoCloser implements AutoCloseable {
055
056    private final Deque<AutoCloseable> stack = new ArrayDeque<>(4);
057
058    private Throwable thrown;
059
060    /**
061     * Registers the given {@code AutoCloseable} resource to be closed when this {@code AutoCloser} is {@link #close
062     * closed}.
063     *
064     * @param <C>      the type of {@code AutoCloseable}
065     * @param resource the given {@code AutoCloseable} resource
066     * @return the given {@code AutoCloseable} resource
067     */
068    public <C extends AutoCloseable> C register(final C resource) {
069        if (resource != null)
070            stack.push(resource);
071        return resource;
072    }
073
074    /**
075     * Stores the given throwable and rethrows it <i>as is</i> if it is an {@code Exception}, {@code RuntimeException} or
076     * {@code Error}. In the unlikely possibility that it is a {@code Throwable} instance, it will be rethrown wrapped in a
077     * {@code RuntimeException}.
078     * <p>
079     * This method always throws and as such should be called as {@code throw closer.rethrow(e)} to ensure the compiler
080     * knows that it will throw.
081     *
082     * @param th the given throwable
083     * @return this method always throws and as such should be called as {@code throw closer.rethrow(e)}
084     * @throws Exception        if the given throwable is an {@code Exception}
085     * @throws Error            if the given throwable is an {@code Error}
086     * @throws RuntimeException if the given throwable is an actual {@code Throwable} instance
087     */
088    public RuntimeException rethrow(final Throwable th) throws Exception {
089        requireNonNull(th, "th == null");
090        thrown = th;
091        throw exception(th);
092    }
093
094    @Override
095    public void close() throws Exception {
096        Throwable th = thrown;
097
098        while (!stack.isEmpty())
099            try {
100                stack.pop().close();
101            } catch (final Throwable e) {
102                if (th == null)
103                    th = e;
104                else
105                    th.addSuppressed(e);
106            }
107
108        if (thrown == null && th != null)
109            throw exception(th);
110    }
111
112    // This method always throws; the return type simply enables the 'throw propagate(Throwable)' syntax
113    private RuntimeException exception(final Throwable th) throws Exception {
114        if (th instanceof Exception)
115            throw (Exception) th;
116        else if (th instanceof Error)
117            throw (Error) th;
118        else
119            throw new RuntimeException(th); // this should never happen unless we actually caught a Throwable instance
120    }
121
122}
